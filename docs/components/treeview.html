Timepicker

_El componente permite mostrar datos en una estructura de árbol

<!--primero se debe crear el elemento-->
<input type="text" class="uif-treeview" id="treeview"/>

Métodos

init

Inicializa el árbol

$("#treeview").UifTreeView();

###selection Type: string

Determina el tipo de seleccion.

<!--primero se debe crear el elemento-->
<input type="text" data-selection="multiple" class="uif-treeview" id="treeview"/>

$("#treeview").UifTreeView({selection: 'single'});//single || multiple

getSelected

Obtiene el valor ingresado en el componente

const selected = $("#treeview").UifTreeView('getSelected');

createNode

selectedNode Type: object

El nodo seleccionado (que será el padre del nuevo nodo)

newNode Type: object

El nuevo nodo

Crea un nodo debajo del nodo seleccionado

const selectedNode = $("#treeview").UifTreeView('getSelected');
if (selectedNode.length > 0) {
    const nodeId = Math.random();
    const newNode =  { id: nodeId, text: 'Nodo de prueba #' + nodeId };
    $("#treeview").UifTreeView('createNode', selectedNode, newNode);
}

renameNode

selectedNode Type: object

El nodo seleccionado a renombrar

newNodeName Type: string

El nuevo nombre para el nodo

Renombra un nodo

const selectedNode = $("#treeview").UifTreeView('getSelected');
if (selectedNode.length > 0) {
    const newName = 'Nuevo nombre de prueba';
    $("#treeview").UifTimepicker('renameNode', selectedNode, newName);
}

renameNode

selectedNode Type: object

El nodo seleccionado a renombrar

newNodeName Type: string

El nuevo nombre para el nodo

Renombra un nodo

const selectedNode = $("#treeview").UifTreeView('getSelected');
if (selectedNode.length > 0) {
    const newName = 'Nuevo nombre de prueba';
    $("#treeview").UifTimepicker('renameNode', selectedNode, newName);

deleteNode

selectedNode Type: object

El nodo seleccionado a eliminar

Elimina un nodo

const selectedNode = $("#treeview").UifTreeView('getSelected');
if (selectedNode.length > 0) {
    $("#treeview").UifTimepicker('deleteNode', selectedNode);
}

reload

Refresca el árbol

$("#treeview").UifTreeView('reload');

setSelected

Selecciona 1 o mas nodos.

$('#tree').UifTreeView('setSelected', "2")
//o
$('#tree').UifTreeView('setSelected', ["2",1])

setUnselect

DeSelecciona 1 o mas nodos.

$('#tree').UifTreeView('setUnselect', "2")
//o
$('#tree').UifTreeView('setUnselect', ["2",1])

unselect

DeSelecciona todos los nodos del arbol.

$('#tree').UifTreeView('unselect')

unselect

Selecciona todos los nodos del arbol(solo esta habilitado para seleccion multiple).

$('#tree').UifTreeView('selectAll')
