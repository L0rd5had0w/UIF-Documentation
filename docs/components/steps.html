Steps

Permite organizar contenidos mediante steps


<div id="steps">
    <div data-title="Primer step">
        <!--contenido del step -->
    </div>
    <div data-title="Segundo step">
        <!--contenido del step -->
    </div>
    <div data-title="Tercer step">
        <!--contenido del step -->
    </div>
</div>

const steps = app.steps({
  container: '#steps',
  onStep: function(event, currentIndex, newIndex){
    return true;
  }
});
