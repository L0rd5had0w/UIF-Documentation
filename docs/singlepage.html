Single Page

Simplificar el manteniendo de la UI, organizando la lógica de manipulación de eventos de una manera clara y simple, cumpliendo un ciclo de vida del componente.

class Person extends Uif2.Page {
    constructor(Person) {
        super();
        this.person = Person;
        //do something
    }
    //Metodo de inicializacion de componentes, ajax, etc.
    getInitialState() {
        $('#select').UifSelect({source: path});
        $('#input').UifAutoComplete({ 
            source: url, 
            displaykey:"Empresa", 
            minLenght: 3, 
            queryParameter: 'query', 
            displayRecords:5 
        });
        this.getAllData();
    }
    //Metodo de asignacion de eventos de los diferentes controles.
    bindEvents() {
        $('#button').on('click', this.searchPerson);
    }

    searchPerson() {
        return this.person;
    }
    getDataPerson() {
        //do something
    }

    static getPerson() {
        return { Name : 'Gabriel'};
    }
}
